### Task: Export Data for Analysis

```bash
# Export all active closures
python cli.py call-tool list_active_entries --limit 1000 > active.json

# Export specific street history
python cli.py call-tool search_obstructions \
  --filters '{"primary_street": "FORBES AVE"}' \
  --limit 5000 > forbes_history.json

# Paginate through all data
for i in {0..5000..500}; do
  python cli.py call-tool search_obstructions \
    --limit 500 --offset $i > batch_$i.json
done
```

**Docs**: [../server/CLI_USAGE.md](../server/CLI_USAGE.md), [../.cursor/skills/domi-obstruction-mcp/examples.md](../.cursor/skills/domi-obstruction-mcp/examples.md)

### Task: Create Custom Automation

```python
#!/usr/bin/env python3
# Example: Daily closure report

import json
import subprocess
from datetime import datetime

result = subprocess.run(
    ['python', 'cli.py', 'call-tool', 'list_active_entries', '--limit', '1000'],
    capture_output=True,
    text=True
)
data = json.loads(result.stdout)

print(f"Daily Closure Report - {datetime.now().strftime('%Y-%m-%d')}")
print(f"Total active closures: {len(data['result']['records'])}")

for record in data['result']['records']:
    print(f"- {record['primary_street']}: {record['work_description']}")
```
